  <action>
    Create `src/components/transactions/TransactionForm.tsx`.

    Props interface:
    ```typescript
    interface TransactionFormProps {
      initialData?: Transaction    // undefined = add mode, Transaction = edit mode
      onSubmit: (data: Omit<Transaction, 'id'>) => void
      onClose: () => void
    }
    ```

    Form state (use a single useState with an object):
    ```typescript
    const [values, setValues] = useState<TransactionFormValues>(() => ({
      type: initialData?.type ?? '',
      category: initialData?.category ?? '',
      amount: initialData ? String(initialData.amount) : '',
      date: initialData?.date ?? '',
      description: initialData?.description ?? '',
    }))
    const [errors, setErrors] = useState<FormErrors>({})
    ```

    Field updater helper (immutable update — never mutate values directly):
    ```typescript
    const update = (field: keyof TransactionFormValues, value: string) =>
      setValues((prev) => ({ ...prev, [field]: value }))
    ```

    Submit handler:
    ```typescript
    const handleSubmit = (e: React.FormEvent) => {
      e.preventDefault()
      const errs = validateTransactionForm(values)
      if (Object.keys(errs).length > 0) {
        setErrors(errs)
        return
      }
      onSubmit({
        type: values.type as TransactionType,
        category: values.category as Category,
        amount: parseFloat(values.amount),
        date: values.date,
        description: values.description,
      })
    }
    ```

    JSX structure: `<form onSubmit={handleSubmit} className="space-y-4">` containing:
    1. Type select (use Select primitive):
       ```tsx
       <Select label="Type" id="txn-type" value={values.type} onChange={(e) => update('type', e.target.value)} error={errors.type}>
         <option value="">Select type</option>
         <option value="income">Income</option>
         <option value="expense">Expense</option>
       </Select>
       ```
    2. Category select (use Select primitive with CATEGORIES):
       ```tsx
       <Select label="Category" id="txn-category" value={values.category} onChange={(e) => update('category', e.target.value)} error={errors.category}>
         <option value="">Select category</option>
         {CATEGORIES.map((c) => <option key={c.id} value={c.id}>{c.label}</option>)}
       </Select>
       ```
    3. Amount input (use Input primitive, type="number", min="0.01", step="0.01"):
       ```tsx
       <Input label="Amount" id="txn-amount" type="number" min="0.01" step="0.01" value={values.amount} onChange={(e) => update('amount', e.target.value)} error={errors.amount} />
       ```
    4. Date input (use Input primitive, type="date"):
       ```tsx
       <Input label="Date" id="txn-date" type="date" value={values.date} onChange={(e) => update('date', e.target.value)} error={errors.date} />
       ```
    5. Description input (use Input primitive):
       ```tsx
       <Input label="Description" id="txn-description" type="text" maxLength={100} value={values.description} onChange={(e) => update('description', e.target.value)} error={errors.description} />
       ```

    6. Action buttons row: `<div className="flex justify-end gap-2 pt-2">`
       - Cancel: `<Button type="button" variant="secondary" onClick={onClose}>Cancel</Button>`
       - Submit: `<Button type="submit">{initialData ? 'Save changes' : 'Add transaction'}</Button>`
